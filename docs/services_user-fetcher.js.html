const axios = require("axios");
const { USER_API } = require("./config/api-fetch-config.js");

/**
 * Fetches user data from the URL.
 *
 * @async
 * @function
 * @returns {Promise} A promise that resolves to the user data.
 * @throws {Error} If there is an error during the fetch.
 */
const getUsers = async function () {
  try {
    const response = await fetch(USER_API.GET_USERS);
    
    if (!response.ok) {
      throw new Error(`HTTP error! Status: ${response.status}`);
    
    }
    
    const data = await response.json();
    return data;
    
  } catch (error) {
    console.error("Error:", error);
    throw error; // Re-throw the error to propagate it
  }
};
